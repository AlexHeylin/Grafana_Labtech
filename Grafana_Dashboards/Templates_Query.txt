Name:  server ($server)
Query: 
SELECT computers.name 
	FROM  Computers 
		LEFT OUTER JOIN clients ON clients.clientid=computers.clientid 
		LEFT OUTER JOIN locations ON locations.locationid=computers.locationid 
	WHERE computers.clientid=clients.clientid AND `OS` LIKE '%server%'
